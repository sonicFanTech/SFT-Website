Build & export workflow

Design → Generate → Export → EXE

How it works

  1. Design your installer wizard pages and UI elements.
  2. The program generates installer code in real time.
  3. Run the generated installer instantly (saved temporarily to a temp file).
  4. Export the installer code so you can build it later.
  5. Use the built-in PyInstaller GUI to compile into an EXE.
  6. Generate + export the uninstaller script, compile it, and bundle it.

def apply_item_style(w, extra: dict):
    """Apply basic widget styling from the designer (fg/bg/font_size/bold)."""
    if not isinstance(extra, dict):
        extra = {}
    fg = str(extra.get("fg", "")).strip()
    bg = str(extra.get("bg", "")).strip()
    fs = extra.get("font_size", None)
    bold = bool(extra.get("bold", False))

    try:
        if isinstance(fs, int) and fs > 0:
            f = w.font(); f.setPointSize(fs); f.setBold(bold); w.setFont(f)
        else:
            f = w.font(); f.setBold(bold); w.setFont(f)
    except Exception:
        pass

    parts = []
    if fg:
        parts.append(f"color: {fg};")
    if bg:
        parts.append(f"background-color: {bg};")
    if parts:
        w.setStyleSheet(" ".join(parts))

def apply_canvas_style(canvas: QWidget, page_extra: dict):
    """Apply page background color/image."""
    if not isinstance(page_extra, dict):
        page_extra = {}
    bg = str(page_extra.get("bg_color", "#141414")).strip() or "#141414"
    img = str(page_extra.get("bg_image", "")).strip()
    if img:
        url = img.replace("\", "/")
        canvas.setStyleSheet(
            f"background-color: {bg};"
            f"background-image: url({url});"
            "background-repeat: no-repeat;"
            "background-position: center;"
        )
    else:
        canvas.setStyleSheet(f"background: {bg};")

def expand_install_dir(s: str) -> str:
    # Expand environment vars like %LOCALAPPDATA%
    if not s:
        return s
    # First expand ~
    s = os.path.expanduser(s)
    # Expand Windows-style %VAR%
    s = os.path.expandvars(s)
    return s

def payload_path() -> str:
    # Prefer bundled files when running from a PyInstaller one-file/one-dir build
    base = getattr(sys, "_MEIPASS", None) or os.path.dirname(os.path.abspath(sys.argv[0]))
    configured = PROJECT.get("payload_zip", "").strip()
    if configured:
        # If user configured an absolute path, prefer that
        cfg = expand_install_dir(configured)
        if os.path.isabs(cfg) and os.path.isfile(cfg):
            return cfg
        # If relative, treat as next to exe
        cand = os.path.join(base, cfg)
        if os.path.isfile(cand):
            return cand
    # default
    cand = os.path.join(base, "payload.zip")
    return cand

class InstallWorker(QThread):
    progress = Signal(int)
    log = Signal(str)
    done = Signal(bool, str)

    def __init__(self, install_dir: str, parent=None):
        super().__init__(parent)
        self.install_dir = install_dir

    def run(self):
        try:
            zip_path = payload_path()
            if not os.path.isfile(zip_path):
                raise FileNotFoundError(f"payload.zip not found: {zip_path}")

            os.makedirs(self.install_dir, exist_ok=True)

            with zipfile.ZipFile(zip_path, "r") as zf:
                names = zf.namelist()
                total = max(1, len(names))
                for i, name in enumerate(names, start=1):
                    zf.extract(name, self.install_dir)
                    pct = int(i * 100 / total)
                    self.progress.emit(pct)
                    self.log.emit(f"Extracted: {name}")

            self.done.emit(True, "Install complete.")
        except Exception as e:
            tb = traceback.format_exc()
            self.done.emit(False, f"Install failed: {e}\n\n{tb}")

class InstallingPage(QWizardPage):
    def __init__(self, spec: dict):
        super().__init__()
        self.spec = spec
        self.setTitle(spec.get("title", "Installing"))
        self.worker = None
        self._started = False

        outer = QVBoxLayout(self)

        canvas = QWidget()
        canvas.setMinimumSize(760, 120)
        apply_canvas_style(canvas, (spec.get("extra", {}) or {}))
        for item in spec.get("items", []):
            w = make_widget(item, canvas)
            w.setGeometry(item.get("x", 20), item.get("y", 20), item.get("w", 180), item.get("h", 28))
            w.show()
        outer.addWidget(canvas)

        self.bar = QProgressBar()
        self.bar.setRange(0, 100)
        outer.addWidget(self.bar)

        self.log = QTextEdit()
        self.log.setReadOnly(True)
        self.log.setMinimumHeight(180)
        outer.addWidget(self.log, 1)

    def initializePage(self):
        # Start install once when page is shown
        if self._started:
            return
        self._started = True

        wiz = self.wizard()
        install_dir = wiz.field("installDir")
        if not install_dir:
            install_dir = os.path.join(os.path.expanduser("~"), "App")

        self.worker = InstallWorker(install_dir)
        self.worker.progress.connect(self.bar.setValue)
        self.worker.log.connect(lambda s: self.log.append(s))
        self.worker.done.connect(self._done)
        self.worker.start()

    def _done(self, ok: bool, msg: str):
        if ok:
            self.log.append("\n" + msg)
            self.wizard().setButtonText(QWizard.NextButton, "Next")
            self.completeChanged.emit()
        else:
            QMessageBox.critical(self, "Install failed", msg)
            self.log.append("\n" + msg)
            self.wizard().reject()

    def isComplete(self) -> bool:
        # Allow Next when progress finished
        return self.bar.value() >= 100

def make_widget(item: dict, parent: QWidget):
    t = item.get("type", "Label")
    text = item.get("text", "")

    if t == "Label":
        w = QLabel(text, parent)
        w.setStyleSheet("color: white;")
        apply_item_style(w, (item.get("extra") or {}))
        return w
    if t == "Button":
        w = QPushButton(text or "Button", parent)
        apply_item_style(w, (item.get("extra") or {}))
        return w
    if t == "CheckBox":
        cb = QCheckBox(text or "CheckBox", parent)
        cb.setChecked(bool((item.get("extra") or {}).get("checked", False)))
        cb.setStyleSheet("color: white;")
        apply_item_style(cb, (item.get("extra") or {}))
        return cb
    if t == "LineEdit":
        le = QLineEdit(parent)
        le.setPlaceholderText(text or "")
        apply_item_style(le, (item.get("extra") or {}))
        return le
    if t == "TextBox":
        te = QTextEdit(parent)
        te.setPlainText(text or "")
        apply_item_style(te, (item.get("extra") or {}))
        return te
    if t == "Image":
        lbl = QLabel(parent)
        lbl.setStyleSheet("color: #cccccc; border: 1px solid #444;")
        lbl.setAlignment(Qt.AlignCenter)
        path = (text or "").strip()
        if path and os.path.exists(path):
            from PySide6.QtGui import QPixmap
            pm = QPixmap(path)
            if not pm.isNull():
                lbl.setPixmap(pm.scaled(item.get("w", 220), item.get("h", 140), Qt.KeepAspectRatio, Qt.SmoothTransformation))
            else:
                lbl.setText("Image load failed")
        else:
            lbl.setText(path or "Image")
        apply_item_style(lbl, (item.get("extra") or {}))
        return lbl

    w = QLabel(text or t, parent)
    w.setStyleSheet("color: white;")
    apply_item_style(w, (item.get("extra") or {}))
    return w

def build_wizard():
    wiz = QWizard()
    wiz.setWindowTitle(f"{PROJECT.get('app_name','App')} Setup")
    wiz.setWizardStyle(QWizard.ModernStyle)

    # Optional dark-ish palette
    wiz.setStyleSheet("""
        QWizard { background: #0f0f0f; }
        QLabel { color: white; }
        QWizardPage { background: #0f0f0f; }
        QTextEdit, QLineEdit { background: #1b1b1b; color: white; border: 1px solid #333; }
        QPushButton { padding: 6px 10px; }
        QProgressBar { height: 16px; }
    """)

    pages = PROJECT.get("pages", [])
    license_text = PROJECT.get("license_text", "")
    default_dir = PROJECT.get("default_install_dir", r"%LOCALAPPDATA%\App")

    for p in pages:
        role = p.get("role", "generic")
        if role == "license":
            wiz.addPage(LicensePage(p, license_text))
        elif role == "install_dir":
            wiz.addPage(InstallDirPage(p, default_dir))
        elif role == "installing":
            wiz.addPage(InstallingPage(p))
        elif role == "finish":
            wiz.addPage(FinishPage(p))
        else:
            wiz.addPage(GenericPage(p))

    # Make sure buttons look sane
    wiz.setOption(QWizard.NoCancelButton, False)
    return wiz

def main():
    app = QApplication(sys.argv)
    wiz = build_wizard()
    wiz.show()
    sys.exit(app.exec())

if __name__ == "__main__":
    main()

FAQ

Common questions

Is this like Inno Setup?

It’s a different approach: Roxzned Setup Designer is a Python-based visual wizard designer that generates runnable Python installer code in real time, then lets you compile it into an EXE using PyInstaller.

Can I customize every page?

Yes — background color, per-page images, text size/color/fonts, object positions, page order/names, and the overall window size are all part of the designer plan.

How does the install path control work?

Add an Edit Box to a page and the designer can auto-add a “Set Path” button and wire the logic automatically. You can also connect that action manually if you want full control.

Does it generate an uninstaller?

Yes — after finishing the main installer design, it auto-generates a separate uninstaller script that you can export, build into an EXE, and package with the main installer.
